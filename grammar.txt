program: <function>*

function: <name> '(' <parameter list> ')' ':' <type or array> '{' <block> '}'

block: '{' <statement>* '}'

return: 'return' <expression> ';'
statement: <return> | <block>

expression: <number> | <unary operator>

unary operator: '-' <expression> | '!' <expression> | ~ <expression>

type mod: 'const'
type: <type mod>* 'int' | 'char'
type or array: <type> | '[' <type> ']'

variable: <name> ':' <type or array>
parameter list: <variable> (',' <variable>)*
